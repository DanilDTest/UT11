								ПолученныеДанные.ДополнительныеСвойства, КомпонентыОбмена);

		ПолученныеДанные.ВалютаДенежныхСредств = (?(ДанныеИБ = Неопределено,
			ВалютаРегламентированногоУчета(КомпонентыОбмена), ДанныеИБ.ВалютаДенежныхСредств));      
	КонецЕсли;

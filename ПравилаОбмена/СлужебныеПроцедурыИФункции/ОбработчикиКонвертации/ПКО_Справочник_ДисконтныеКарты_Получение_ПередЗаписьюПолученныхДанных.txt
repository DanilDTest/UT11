Процедура ПКО_Справочник_ДисконтныеКарты_Получение_ПередЗаписьюПолученныхДанных(ПолученныеДанные, ДанныеИБ, КонвертацияСвойств, КомпонентыОбмена)
	Если НЕ ДанныеИБ = Неопределено Тогда
		ПолученныеДанные.Владелец = ДанныеИБ.Владелец;
	
		ТипКарты = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ДанныеИБ.Владелец, "ТипКарты");
		Если ТипКарты = Перечисления.ТипыКарт.Штриховая Тогда
			ДанныеИБ.Штрихкод = ПолученныеДанные.Штрихкод;
		ИначеЕсли ТипКарты = Перечисления.ТипыКарт.Магнитная Тогда
			ДанныеИБ.МагнитныйКод = ПолученныеДанные.МагнитныйКод;
		ИначеЕсли НЕ (ДанныеИБ.Штрихкод = ПолученныеДанные.Штрихкод ИЛИ ДанныеИБ.МагнитныйКод = ПолученныеДанные.МагнитныйКод) Тогда
			ДанныеИБ.Штрихкод     = ПолученныеДанные.Штрихкод;
			ДанныеИБ.МагнитныйКод = ПолученныеДанные.МагнитныйКод;
		КонецЕсли;
	КонецЕсли;
КонецПроцедуры

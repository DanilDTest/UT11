Процедура ПОД_Справочник_Банки_Получение_ПриОбработке(ДанныеXDTO, ИспользованиеПКО, КомпонентыОбмена)
	Если НЕ ДанныеXDTO.КлючевыеСвойства.Свойство("БИК")
		ИЛИ (ОбщегоНазначения.РазделениеВключено()
		И ОбщегоНазначения.ДоступноИспользованиеРазделенныхДанных()) Тогда
		
		ОчиститьИспользованиеПКО(ИспользованиеПКО);
	
		// Если у банка не заполнен БИК или используется разделение данных (и банк не найден),
		// реквизиты банков загружаются непосредственно в банковские счетах. 	
		ТаблицаБанки = КомпонентыОбмена.ПараметрыКонвертации.ТаблицаБанки;
		
		СтрокаБанки = ТаблицаБанки.Добавить();
		ЗаполнитьЗначенияСвойств(СтрокаБанки, ДанныеXDTO);
		ЗаполнитьЗначенияСвойств(СтрокаБанки, ДанныеXDTO.КлючевыеСвойства);
		
	КонецЕсли;
КонецПроцедуры

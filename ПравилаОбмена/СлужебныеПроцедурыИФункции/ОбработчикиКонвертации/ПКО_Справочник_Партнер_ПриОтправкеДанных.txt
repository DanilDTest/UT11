Процедура ПКО_Справочник_Партнер_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	ПодготовитьДанныеКонтактнаяИнформация(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена);  
	
	СтруктураДополнительныхРеквизитов = Новый Структура;
	СтруктураДополнительныхРеквизитов.Вставить("НаименованиеПолное", ДанныеИБ.НаименованиеПолное);
	СтруктураДополнительныхРеквизитов.Вставить("ДатаРегистрации", ДанныеИБ.ДатаРегистрации);
	СтруктураДополнительныхРеквизитов.Вставить("Клиент", ДанныеИБ.Клиент);
	СтруктураДополнительныхРеквизитов.Вставить("Поставщик", ДанныеИБ.Поставщик);
	СтруктураДополнительныхРеквизитов.Вставить("ПрочиеОтношения", ДанныеИБ.ПрочиеОтношения);
	СтруктураДополнительныхРеквизитов.Вставить("Перевозчик", ДанныеИБ.Перевозчик);
	СтруктураДополнительныхРеквизитов.Вставить("Комментарий", ДанныеИБ.Комментарий);
	СтруктураДополнительныхРеквизитов.Вставить("Конкурент", ДанныеИБ.Конкурент);
	СтруктураДополнительныхРеквизитов.Вставить("ОбслуживаетсяТорговымиПредставителями", ДанныеИБ.ОбслуживаетсяТорговымиПредставителями);
	
	ЗаполнитьЗначенияСвойств(СтруктураДополнительныхРеквизитов, ДанныеИБ);  
	ВыгрузитьДополнительныеРеквизитыСвойствоСтрокой(СтруктураДополнительныхРеквизитов, ДанныеXDTO);

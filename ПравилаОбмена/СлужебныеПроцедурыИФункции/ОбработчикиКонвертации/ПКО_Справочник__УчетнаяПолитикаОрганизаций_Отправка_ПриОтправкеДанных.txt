Процедура ПКО_Справочник__УчетнаяПолитикаОрганизаций_Отправка_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	ДанныеXDTO.Вставить("Организация", ДанныеИБ.Отбор.Организация.Значение);
	ДанныеXDTO.Вставить("Период", ДанныеИБ.Отбор.Период.Значение);
	
	ТаблицаЗаписи = ДанныеИБ.Выгрузить();
	
	ДанныеXDTO.Вставить("Записи", ТаблицаЗаписи);
	                                                 
	УП = Неопределено;
	Для каждого Запись из ТаблицаЗаписи Цикл
		УП = Запись.УчетнаяПолитика;
	КонецЦикла;  
	
	Если ЗначениеЗаполнено(УП) Тогда  
		ПравилоУчетныеПолитики = КомпонентыОбмена.ПравилаОбработкиДанных.Найти("Справочник_УчетныеПолитикиОрганизаций_Отправка", "Имя");                                             
		RM_ОбменДаннымиXDTOСервер.ВыгрузкаОбъектаВыборки(КомпонентыОбмена, УП, ПравилоУчетныеПолитики); 
	КонецЕсли;
КонецПроцедуры

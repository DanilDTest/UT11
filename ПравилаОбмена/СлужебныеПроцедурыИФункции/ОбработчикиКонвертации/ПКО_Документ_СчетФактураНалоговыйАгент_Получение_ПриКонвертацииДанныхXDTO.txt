Процедура ПКО_Документ_СчетФактураНалоговыйАгент_Получение_ПриКонвертацииДанныхXDTO(ДанныеXDTO, ПолученныеДанные, КомпонентыОбмена)
	ПолученныеДанные.Проведен = Истина;
	ПолученныеДанные.ХозяйственнаяОперация = Перечисления.ХозяйственныеОперации.НачислениеНДСНалоговымАгентом;
	
	ЗагрузитьОтветственноеЛицо(ПолученныеДанные, ДанныеXDTO, КомпонентыОбмена, "ГлавныйБухгалтер");
	ЗагрузитьОтветственноеЛицо(ПолученныеДанные, ДанныеXDTO, КомпонентыОбмена, "Руководитель");
	
	ДокОснование = ЗначениеВИнструкциюПКО(КомпонентыОбмена, ДанныеXDTO.ДокументыОснования[0].ДокументОснование);
	ПолученныеДанные.ДополнительныеСвойства.Вставить("ДокументОснование", ДокОснование);
	
	#Область РасшифровкаСуммы
	
	СтруктураДанныхСтроки = Новый Структура;
	Если ДанныеXDTO.Свойство("СтавкаНДС") И ЗначениеЗаполнено(ДанныеXDTO.СтавкаНДС) Тогда
		СтруктураДанныхСтроки.Вставить("СтавкаНДС", ДанныеXDTO.СтавкаНДС);
	КонецЕсли;
	СуммаСНДС = 0;
	Если ДанныеXDTO.Свойство("Сумма") Тогда
		СтруктураДанныхСтроки.Вставить("Сумма", ДанныеXDTO.Сумма);
		СуммаСНДС = СуммаСНДС + ДанныеXDTO.Сумма; 
	КонецЕсли;
	Если ДанныеXDTO.Свойство("СуммаНДС") Тогда
		СтруктураДанныхСтроки.Вставить("СуммаНДС", ДанныеXDTO.СуммаНДС);
		СуммаСНДС = СуммаСНДС + ДанныеXDTO.СуммаНДС;
	КонецЕсли;
	Если НЕ СуммаСНДС = 0 Тогда
		СтруктураДанныхСтроки.Вставить("СуммаСНДС", СуммаСНДС);
	КонецЕсли;
	
	Если ЗначениеЗаполнено(СтруктураДанныхСтроки) Тогда
		МассивРасшифровкаСуммы = Новый Массив;
		МассивРасшифровкаСуммы.Добавить(СтруктураДанныхСтроки);
		
		ПолученныеДанные.ДополнительныеСвойства.Вставить("РасшифровкаСуммы", МассивРасшифровкаСуммы);
	КонецЕсли;
	
	#КонецОбласти
	
	ЗагрузитьДополнительныеРеквизиты(ПолученныеДанные, ДанныеXDTO, КомпонентыОбмена);
КонецПроцедуры

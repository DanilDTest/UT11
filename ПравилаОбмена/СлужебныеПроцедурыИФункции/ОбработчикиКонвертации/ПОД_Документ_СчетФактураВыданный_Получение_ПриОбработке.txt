Процедура ПОД_Документ_СчетФактураВыданный_Получение_ПриОбработке(ДанныеXDTO, ИспользованиеПКО, КомпонентыОбмена)
	УстановитьИспользованиеПКО(ИспользованиеПКО, Ложь);
	
	Если ДанныеXDTO.Свойство("Авансы") Тогда
		ИспользованиеПКО.Документ_СчетФактураВыданныйАванс_Получение = Истина;
	ИначеЕсли ДанныеXDTO.Свойство("НомерСчетаФактурыКомиссионера")
		И ЗначениеЗаполнено(ДанныеXDTO.НомерСчетаФактурыКомиссионера)
		И ДанныеXDTO.ДокументыОснования.Количество() = 1 Тогда
		ИспользованиеПКО.Документ_СчетФактураКомиссионеру_Получение = Истина;
	ИначеЕсли ДанныеXDTO.ВидСчетаФактуры.Значение = "НалоговыйАгент" Тогда
		ИспользованиеПКО.Документ_СчетФактураНалоговыйАгент_Получение = Истина;
	Иначе
		ИспользованиеПКО.Документ_СчетФактураВыданный_Получение = Истина;
	КонецЕсли;
КонецПроцедуры

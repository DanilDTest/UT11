Процедура ПКО_Справочник_ПодарочныеСертификаты_ПередЗаписьюПолученныхДанных(ПолученныеДанные, ДанныеИБ, КонвертацияСвойств, КомпонентыОбмена)
	Если НЕ ЗначениеЗаполнено(ПолученныеДанные.Наименование) Тогда
		СтрокаНаименования = Строка(ПолученныеДанные.Владелец)
			                     + ?(ЗначениеЗаполнено(ПолученныеДанные.Штрихкод), " " + Строка(ПолученныеДанные.Штрихкод), "")
			                     + ?(ЗначениеЗаполнено(ПолученныеДанные.МагнитныйКод), " " + Строка(ПолученныеДанные.МагнитныйКод), "");
	
		ПолученныеДанные.Наименование = СтрокаНаименования;
	КонецЕсли;

Процедура ПОД_Справочник_ДоговорыМеждуОрганизациями_ПриОбработке(ДанныеИБ, ИспользованиеПКО, КомпонентыОбмена)
	ПравилоДоговорыИзСтруктуры = КомпонентыОбмена.ПравилаОбработкиДанных.Найти("Справочник_ДоговорыМеждуОрганизациями", "Имя");
	
	Если ПравилоДоговорыИзСтруктуры = Неопределено Тогда
		Возврат;
	КонецЕсли;
	
	Если ТипЗнч(ДанныеИБ) <> Тип("Структура") Тогда	
		ДанныеДоговора = ДанныеДоговораМеждуОрганизациями(КомпонентыОбмена, ДанныеИБ);
		RM_ОбменДаннымиXDTOСервер.ВыгрузкаОбъектаВыборки(КомпонентыОбмена, ДанныеДоговора, ПравилоДоговорыИзСтруктуры);
	Иначе
		ОчиститьИспользованиеПКО(ИспользованиеПКО);
		ИспользованиеПКО.Вставить("Справочник_ДоговорыКонтрагентов_ИзСтруктуры", Истина);
	КонецЕсли;
КонецПроцедуры

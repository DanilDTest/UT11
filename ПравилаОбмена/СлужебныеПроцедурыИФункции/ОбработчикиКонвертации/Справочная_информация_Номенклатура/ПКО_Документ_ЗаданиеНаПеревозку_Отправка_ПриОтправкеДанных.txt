Процедура ПКО_Документ_ЗаданиеНаПеревозку_Отправка_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	Приоритет = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ДанныеИБ.Приоритет, "Наименование");  
	НаименованиеГруза = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ДанныеИБ.RM_НаименованиеГруза, "Наименование");  
	
	ДанныеXDTO.Вставить("Приоритет", Приоритет);        
	ДанныеXDTO.Вставить("НаименованиеГруза", НаименованиеГруза);    
	     
	ТипТранспорта = ТипЗнч(ДанныеИБ.ТранспортноеСредство);     
	Если ТипТранспорта = Тип("СправочникСсылка.ТипыТранспортныхСредств") Тогда
		ИмяПКО = "ТипыТранспортныхСредств";
	Иначе
		ИмяПКО = "Справочник_ТранспортныеСредства";
	КонецЕсли;
	ДанныеXDTO.Вставить("ТранспортноеСредство", Новый Структура("Значение, ИмяПКО", ДанныеИБ.ТранспортноеСредство, ИмяПКО));  
	
	
	
	Маршрут = ДанныеИБ.Маршрут.Выгрузить();
	ДанныеXDTO.Вставить("Маршрут", Маршрут);
	   
	УпаковочныеЛисты = ДанныеИБ.RM_УпаковочныеЛисты.Выгрузить();
	ДанныеXDTO.Вставить("УпаковочныеЛисты", УпаковочныеЛисты);
	 
	ВложенныеЗадания = ДанныеИБ.RM_ВложенныеЗадания.Выгрузить();
	ДанныеXDTO.Вставить("ВложенныеЗадания", ВложенныеЗадания);
КонецПроцедуры

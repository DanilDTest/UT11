Процедура ПКО_Документ_ПервичныйДокумент_Отправка_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	Контрагент = ДанныеИБ.Контрагент;
	Если ТипЗнч(Контрагент) = Тип("СправочникСсылка.Организации") Тогда
		Контрагент = КонтрагентИзОрганизации(Контрагент, КомпонентыОбмена);
	КонецЕсли;
	ДанныеXDTO.КлючевыеСвойства.Вставить("Контрагент",  Контрагент);
	
	Если ЗначениеЗаполнено(ДанныеИБ.Договор) И ТипЗнч(ДанныеИБ.Договор) = Тип("СправочникСсылка.ДоговорыКонтрагентов") Тогда
		ДанныеXDTO.Вставить("Договор",  ДанныеИБ.Договор);
	КонецЕсли;
КонецПроцедуры

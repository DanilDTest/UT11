Процедура ПКО_Справочник_КассыККМ_Отправка_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	Если ВерсияФорматаЧислом(КомпонентыОбмена.ВерсияФорматаОбмена) >= ВерсияФорматаЧислом("1.7") Тогда
		ПараметрыПолитики = РегистрыСведений.УчетнаяПолитикаОрганизаций.ПараметрыУчетнойПолитики(ДанныеИБ.Владелец);
		ПараметрыУчетаПоОрганизации = УчетНДСУП.ПараметрыУчетаПоОрганизации(ДанныеИБ.Владелец,
			ТекущаяДатаСеанса(),
			ДанныеИБ.Склад);
		Если НЕ ПараметрыПолитики = Неопределено Тогда
			Если ПараметрыПолитики.ПрименяетсяЕНВД И ПараметрыУчетаПоОрганизации.ПрименяетсяЕНВД Тогда
				СистемаНалогообложения = "ЕНВД";
			ИначеЕсли ПараметрыПолитики.ПрименяетсяУСН Тогда
				СистемаНалогообложения = "УпрощеннаяДоход";
			ИначеЕсли ПараметрыПолитики.ПрименяетсяУСНДоходыМинусРасходы Тогда
				СистемаНалогообложения = "УпрощеннаяДоходМинусРасход";
			Иначе
				СистемаНалогообложения = "Общая";
			КонецЕсли;
			
			ДанныеXDTO.Вставить("СистемаНалогообложения", СистемаНалогообложения);
		КонецЕсли;
	КонецЕсли;
	
	Если ВерсияФорматаЧислом(КомпонентыОбмена.ВерсияФорматаОбмена) >= ВерсияФорматаЧислом("1.8") Тогда
		ВыгрузитьДополнительныеРеквизитыИСведения(КомпонентыОбмена, ДанныеИБ, ДанныеXDTO, Истина);
	КонецЕсли;
КонецПроцедуры

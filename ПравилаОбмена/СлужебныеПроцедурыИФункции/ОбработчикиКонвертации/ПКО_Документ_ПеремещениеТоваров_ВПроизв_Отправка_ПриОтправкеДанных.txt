Процедура ПКО_Документ_ПеремещениеТоваров_ВПроизв_Отправка_ПриОтправкеДанных(ДанныеИБ, ДанныеXDTO, КомпонентыОбмена, СтекВыгрузки)
	НомерДок = Лев(ДанныеИБ.Номер, 5) + "П" + Сред(ДанныеИБ.Номер, 7);
	
	Если СтекВыгрузки.Количество() > 1 Тогда
		ДанныеXDTO.Вставить("Номер", НомерДок);
		Возврат;
	КонецЕсли;
	
	ДанныеXDTO.КлючевыеСвойства.Вставить("Номер", НомерДок);
	
	СкладДляВыгрузки = СкладДляВыгрузки(ДанныеИБ.СкладОтправитель, КомпонентыОбмена.ПараметрыКонвертации);
	ДанныеXDTO.Вставить("Склад", СкладДляВыгрузки);
	ВыгрузитьПодразделение(ДанныеИБ, ДанныеXDTO, ОбщегоНазначения.ЗначениеРеквизитаОбъекта(ДанныеИБ.СкладПолучатель, "Подразделение"));
	ВыгрузитьНалогообложениеНДС(КомпонентыОбмена, ДанныеИБ, ДанныеXDTO, Ложь, Ложь);
	
	ВыгрузитьДополнительныеРеквизитыИСведения(КомпонентыОбмена, ДанныеИБ, ДанныеXDTO);
	ДанныеXDTO.Вставить("Товары", ДанныеИБ.ДополнительныеСвойства.Товары);
КонецПроцедуры

Процедура ПОД_Документ_СчетФактураПолученный_Получение_ПриОбработке(ДанныеXDTO, ИспользованиеПКО, КомпонентыОбмена)
	УстановитьИспользованиеПКО(ИспользованиеПКО, Ложь);
	
	Если ДанныеXDTO.Свойство("Авансы") Тогда
		ИспользованиеПКО.Документ_СФактураПолученныйАванс_Получение = Истина;
	ИначеЕсли ДанныеXDTO.Свойство("СчетаФактурыВыданныеПокупателям") Тогда
		ИспользованиеПКО.Документ_СчетФактураКомитента_Получение = Истина;
	ИначеЕсли ДанныеXDTO.ВидСчетаФактуры.Значение = "НалоговыйАгент" Тогда
		ИспользованиеПКО.Документ_СчетФактураПолученныйНалоговыйАгент_Получ = Истина;
	Иначе
		ИспользованиеПКО.Документ_СчетФактураПолученный_Получение = Истина;
	КонецЕсли;
КонецПроцедуры

Процедура ПКО_Справочник_Упаковки_ПередЗаписьюПолученныхДанных(ПолученныеДанные, ДанныеИБ, КонвертацияСвойств, КомпонентыОбмена)
	Если НЕ ЗначениеЗаполнено(ПолученныеДанные.Владелец)
		ИЛИ (НЕ ДанныеИБ = Неопределено И ТипЗнч(ДанныеИБ.Владелец) = Тип("СправочникСсылка.НаборыУпаковок")) Тогда
		ПолученныеДанные = Неопределено;
	КонецЕсли;
	
	ДанныеДляЗаписиВИБ = ?(ДанныеИБ = Неопределено, ПолученныеДанные, ДанныеИБ);
	Если Не ЗначениеЗаполнено(ДанныеДляЗаписиВИБ.Знаменатель) Тогда
		ДанныеДляЗаписиВИБ.Знаменатель = 1;
	КонецЕсли;
	Если Не ЗначениеЗаполнено(ДанныеДляЗаписиВИБ.ТипУпаковки) Тогда
		ДанныеДляЗаписиВИБ.ТипУпаковки = Перечисления.ТипыУпаковокНоменклатуры.Конечная;
	КонецЕсли;
	Если Не ЗначениеЗаполнено(ДанныеДляЗаписиВИБ.ТипИзмеряемойВеличины) Тогда
		ДанныеДляЗаписиВИБ.ТипИзмеряемойВеличины = Перечисления.ТипыИзмеряемыхВеличин.Упаковка;
	КонецЕсли;
КонецПроцедуры

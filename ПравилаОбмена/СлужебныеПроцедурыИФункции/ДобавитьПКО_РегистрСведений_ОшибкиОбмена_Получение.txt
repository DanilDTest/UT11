Процедура ДобавитьПКО_РегистрСведений_ОшибкиОбмена_Получение(ПравилаКонвертации)
	
	ПравилоКонвертации = ОбменДаннымиXDTOСервер.ИнициализироватьПравилоКонвертацииОбъекта(ПравилаКонвертации);
	ПравилоКонвертации.ИмяПКО                       = "РегистрСведений_ОшибкиОбмена_Получение";
	ПравилоКонвертации.ОбъектДанных                 = Метаданные.РегистрыСведений.RM_ОшибкиОбмена;
	ПравилоКонвертации.ОбъектФормата                = "Справочник.RM_ОшибкиОбмена";
	ПравилоКонвертации.ПриКонвертацииДанныхXDTO     = "ПКО_РегистрСведений_ОшибкиОбмена_Получение_ПриКонвертацииДанныхXDTO";
	ПравилоКонвертации.ПередЗаписьюПолученныхДанных = "ПКО_РегистрСведений_ОшибкиОбмена_Получение_ПередЗаписьюПолученныхДанных";
	ПравилоКонвертации.ВариантИдентификации         = "ПоПолямПоиска";
	СвойстваШапки = ПравилоКонвертации.Свойства;
	ДобавитьПКС(СвойстваШапки, "", "БазаВозникновенияОшибки", , "Справочник_RM_УзлыОбмена");
	ДобавитьПКС(СвойстваШапки, "", "БазаИсточникСообщения", , "Справочник_RM_УзлыОбмена");
	ДобавитьПКС(СвойстваШапки, "", "УниверсальнаяДатаОшибки", 1);

	ПравилоКонвертации.ПоляПоиска.Добавить("БазаВозникновенияОшибки,БазаИсточникСообщения,УниверсальнаяДатаОшибки");

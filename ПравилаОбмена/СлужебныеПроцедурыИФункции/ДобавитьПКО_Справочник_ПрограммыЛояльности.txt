	ПравилоКонвертации.ОбъектФормата                = "Справочник.ПрограммыЛояльности";
	ПравилоКонвертации.ПриОтправкеДанных            = "ПКО_Справочник_ПрограммыЛояльности_ПриОтправкеДанных";

	ПравилоКонвертации.ПередЗаписьюПолученныхДанных = "ПКО_Справочник_ПрограммыЛояльности_ПередЗаписьюПолученныхДанных";

	ДобавитьПКС(СвойстваШапки, "ВалютаКонвертацииБонусов",          "", 1);
	ДобавитьПКС(СвойстваШапки, "КурсКонвертацииБонусовВВалюту",     "КурсКонвертацииБонусовВВалюту");

	ДобавитьПКС(СвойстваШапки, "Наименование",                      "Наименование");
	ДобавитьПКС(СвойстваШапки, "НеНачислятьБаллыПриОплатеБонусами", "НеНачислятьБаллыПриОплатеБонусами");
	
	СвойстваТЧ = ДобавитьПКТЧ(ПравилоКонвертации, "",              "ДополнительныеРеквизиты");
	ДобавитьПКС(СвойстваТЧ, "", "ЗначениеСвойства", 1);
	ДобавитьПКС(СвойстваТЧ, "", "Свойство", 1, "Справочник_ДополнительныеРеквизиты");
	
	СвойстваТЧ = ДобавитьПКТЧ(ПравилоКонвертации, "ЦеновыеГруппы", "ЦеновыеГруппы");
	ДобавитьПКС(СвойстваТЧ, "МаксимальныйПроцентОплатыБонусами", "МаксимальныйПроцентОплатыБонусами");
	ДобавитьПКС(СвойстваТЧ, "ЦеноваяГруппа",                     "ЦеноваяГруппа", , "Справочник_ЦеновыеГруппы");

	ПравилоКонвертации.ПоляПоиска.Добавить("ВалютаКонвертацииБонусов,Наименование");

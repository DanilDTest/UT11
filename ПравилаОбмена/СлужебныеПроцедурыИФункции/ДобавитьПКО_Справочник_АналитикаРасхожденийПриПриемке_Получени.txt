Процедура ДобавитьПКО_Справочник_АналитикаРасхожденийПриПриемке_Получени(ПравилаКонвертации)
	
	ПравилоКонвертации = ОбменДаннымиXDTOСервер.ИнициализироватьПравилоКонвертацииОбъекта(ПравилаКонвертации);
	ПравилоКонвертации.ИмяПКО                       = "Справочник_АналитикаРасхожденийПриПриемке_Получени";
	ПравилоКонвертации.ОбъектДанных                 = Метаданные.РегистрыСведений.RM_АналитикаРасхожденийПриПриемке;
	ПравилоКонвертации.ОбъектФормата                = "Справочник.АналитикаРасхожденийПоПриемке";
	ПравилоКонвертации.ПриКонвертацииДанныхXDTO     = "ПКО_Справочник_АналитикаРасхожденийПриПриемке_Получени_ПриКонвертацииДанныхXDTO";
	ПравилоКонвертации.ПередЗаписьюПолученныхДанных = "ПКО_Справочник_АналитикаРасхожденийПриПриемке_Получени_ПередЗаписьюПолученныхДанных";
	ПравилоКонвертации.ВариантИдентификации         = "ПоПолямПоиска";
	СвойстваШапки = ПравилоКонвертации.Свойства;
	ДобавитьПКС(СвойстваШапки, "", "ПеремещениеТоваров", 1, "Документ_ПеремещениеТоваров_Получение");
	ДобавитьПКС(СвойстваШапки, "", "Склад", 1, "Справочник_Склады_Получение");
	ДобавитьПКС(СвойстваШапки, "", "УпаковочныйЛист", 1, "Документ_УпаковочныйЛист_Получение");

	ПравилоКонвертации.ПоляПоиска.Добавить("Номенклатура,ПеремещениеТоваров,Склад,УпаковочныйЛист");

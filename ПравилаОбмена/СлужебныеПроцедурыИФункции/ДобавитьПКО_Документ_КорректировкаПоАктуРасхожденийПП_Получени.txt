Процедура ДобавитьПКО_Документ_КорректировкаПоАктуРасхожденийПП_Получени(ПравилаКонвертации)
	
	ПравилоКонвертации = ОбменДаннымиXDTOСервер.ИнициализироватьПравилоКонвертацииОбъекта(ПравилаКонвертации);
	ПравилоКонвертации.ИмяПКО                       = "Документ_КорректировкаПоАктуРасхожденийПП_Получени";
	ПравилоКонвертации.ОбъектДанных                 = Метаданные.Документы.RM_КорректировкаПоАктуОРасхожденияхПослеПеремещения;
	ПравилоКонвертации.ОбъектФормата                = "Документ.КорректировкаПоАктуРасхожденийПослеПеремещения";
	ПравилоКонвертации.ВариантИдентификации         = "СначалаПоУникальномуИдентификаторуПотомПоПолямПоиска";
	СвойстваШапки = ПравилоКонвертации.Свойства;
	ДобавитьПКС(СвойстваШапки, "АктОРасхожденияхПослеПеремещения", "АктОРасхожденияхПослеПеремещения", , "Документ_АктОРасхожденияхПослеПеремещения_Получени");
	ДобавитьПКС(СвойстваШапки, "Дата",                             "Дата");
	ДобавитьПКС(СвойстваШапки, "ЗаказНаПеремещение",               "ЗаказНаПеремещениеСсылка", , "Документ_ЗаказНаПеремещение_Получение");
	ДобавитьПКС(СвойстваШапки, "Номер",                            "Номер");
	ДобавитьПКС(СвойстваШапки, "Организация",                      "Организация", , "Справочник_Организации_Получение");
	ДобавитьПКС(СвойстваШапки, "ПеремещениеТоваров",               "ПеремещениеТоваров", , "Документ_ПеремещениеТоваров_Получение");
	ДобавитьПКС(СвойстваШапки, "Проведен",                         "Проведен");
	
	СвойстваТЧ = ДобавитьПКТЧ(ПравилоКонвертации, "Товары", "Товары");
	ДобавитьПКС(СвойстваТЧ, "ИзлишекНеПризнан", "ИзлишекНеПризнан");
	ДобавитьПКС(СвойстваТЧ, "Количество",       "Количество");
	ДобавитьПКС(СвойстваТЧ, "Номенклатура",     "Номенклатура", , "Справочник_Номенклатура_Получение");
	ДобавитьПКС(СвойстваТЧ, "Характеристика",   "Характеристика", , "Справочник_ХарактеристикиНоменклатуры");

	ПравилоКонвертации.ПоляПоиска.Добавить("Дата,Номер");

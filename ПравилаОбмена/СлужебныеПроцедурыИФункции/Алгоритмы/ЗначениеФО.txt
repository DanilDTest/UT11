Функция ЗначениеФО(ИмяФО, КомпонентыОбмена)
	ЗначенияФО = КомпонентыОбмена.ПараметрыКонвертации.ЗначенияФО;
	
	ЗначениеФО = ЗначенияФО.Получить(ИмяФО);
	Если ЗначениеФО = Неопределено Тогда
		ЗначениеФО = ПолучитьФункциональнуюОпцию(ИмяФО);
		ЗначенияФО.Вставить(ИмяФО, ЗначениеФО);
	КонецЕсли;
	
	Возврат ЗначениеФО;
КонецФункции

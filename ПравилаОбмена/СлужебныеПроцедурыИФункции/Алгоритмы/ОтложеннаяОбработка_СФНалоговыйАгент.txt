Процедура ОтложеннаяОбработка_СФНалоговыйАгент(Объект, КомпонентыОбмена)
	Если ЗначениеЗаполнено(Объект.ДокументОснование)
		И ОбщегоНазначения.СсылкаСуществует(Объект.ДокументОснование) Тогда
			
		Объект.Подразделение = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(Объект.ДокументОснование,
			"Подразделение");
		
	КонецЕсли;
	
	// Заполнение партнера.
	Партнер = ДенежныеСредстваСервер.ПолучитьПартнераПоКонтрагенту(Объект.Поставщик);
	Если Не ЗначениеЗаполнено(Партнер) Тогда
		Партнер = Справочники.Партнеры.НеизвестныйПартнер;
	КонецЕсли;
	
	Если Не ЗначениеЗаполнено(Объект.Партнер)
		Или Объект.Партнер <> Партнер Тогда
		Объект.Партнер = Партнер;	
	КонецЕсли;

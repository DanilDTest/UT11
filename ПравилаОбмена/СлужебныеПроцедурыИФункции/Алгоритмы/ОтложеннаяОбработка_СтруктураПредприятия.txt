Процедура ОтложеннаяОбработка_СтруктураПредприятия(Объект)
	МетаданныеОбъекта = Объект.Метаданные();
	
	Если Не ЗначениеЗаполнено(Объект.ВариантОбособленногоУчетаТоваров) Тогда	
		Объект.ВариантОбособленногоУчетаТоваров = Перечисления.ВариантыОбособленногоУчетаТоваров.НеВедется;
	КонецЕсли;
	
	Если (Объект.ПроизводствоПоЗаказам ИЛИ Объект.ПроизводствоБезЗаказов)
		И Не ЗначениеЗаполнено(Объект.СпособНастройкиГрафикаРаботы) Тогда 
						
		Если НЕ ЗначениеЗаполнено(Объект.ГрафикРаботы) И Объект.ГрафикиРаботы.Количество() = 0 Тогда
			Объект.СпособНастройкиГрафикаРаботы = Перечисления.СпособыНастройкиГрафикаРаботыПодразделений.ОсновнойКалендарьПредприятия;
		Иначе
			Объект.СпособНастройкиГрафикаРаботы = Перечисления.СпособыНастройкиГрафикаРаботыПодразделений.ИндивидуальныйГрафик;
			Если ЗначениеЗаполнено(Объект.ГрафикРаботы) И Объект.ГрафикиРаботы.Количество() = 0 Тогда
				Объект.ГрафикиРаботы.Добавить().ГрафикРаботы = Объект.ГрафикРаботы;
				Объект.ГрафикРаботы = Неопределено;
			КонецЕсли;
		КонецЕсли;		
	КонецЕсли;

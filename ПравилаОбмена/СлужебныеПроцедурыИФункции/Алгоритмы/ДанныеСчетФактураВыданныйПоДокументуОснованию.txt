Функция ДанныеСчетФактураВыданныйПоДокументуОснованию(ДокументОснование)
	Результат = Новый Структура;
	Результат.Вставить("ПредставлениеНомера");
	Результат.Вставить("Дата");
	Результат.Вставить("Номер");
	Результат.Вставить("Ссылка");
	
	
	Запрос = Новый Запрос;
	Запрос.Текст = "ВЫБРАТЬ ПЕРВЫЕ 1
	|	СчетФактураВыданный.Ссылка КАК Ссылка,
	|	СчетФактураВыданный.ПредставлениеНомера КАК ПредставлениеНомера,
	|	СчетФактураВыданный.Дата КАК Дата,
	|	СчетФактураВыданный.Номер КАК Номер
	|ИЗ
	|	Документ.СчетФактураВыданный КАК СчетФактураВыданный
	|ГДЕ
	|	СчетФактураВыданный.ДокументОснование = &ДокументОснование";
	//|	И СчетФактураВыданный.Проведен";
	
	Запрос.УстановитьПараметр("ДокументОснование", ДокументОснование);
	РезультатЗапроса = Запрос.Выполнить();
	
	Если НЕ РезультатЗапроса.Пустой() Тогда
		
		Выборка = РезультатЗапроса.Выбрать();
		Выборка.Следующий(); 
		
		Результат.Вставить("ПредставлениеНомера", Выборка.ПредставлениеНомера);
		Результат.Вставить("Дата", Выборка.Дата);
		Результат.Вставить("Номер", Выборка.Номер);
		Результат.Вставить("Ссылка", Выборка.Ссылка);
		
	КонецЕсли;
	
	Возврат Результат;

Функция ПолучитьОбщиеСведения(ДанныеXDTO, ПолученныеДанные = Неопределено, КомпонентыОбмена = Неопределено)
	ОбщиеСведения = Новый Структура; 
		
	ДополнительныеРеквизиты = Неопределено;
	Если Не ДанныеXDTO.Свойство("ДополнительныеРеквизиты", ДополнительныеРеквизиты) Тогда
		Возврат ОбщиеСведения; 
	КонецЕсли; 
	
	МассивСтрокДляУдаления = Новый Массив;
	ПрефиксНаименования = ПрефиксСвойстваОбщееСведения();
	
	Для Каждого СтрокаДопРеквизита Из ДополнительныеРеквизиты Цикл 
		  
		НаименованиеСвойства = СтрокаДопРеквизита.Свойство.Наименование;	  
		Если НЕ СтрНайти(НаименованиеСвойства, ПрефиксНаименования) = 0 Тогда 
			МассивСтрокДляУдаления.Добавить(СтрокаДопРеквизита);
			
			НаименованиеРеквизита = СтрЗаменить(НаименованиеСвойства, ПрефиксНаименования, "");
			ЗначениеРеквизита = СтрокаДопРеквизита.ЗначениеСвойства;
			
			ОбщиеСведения.Вставить(НаименованиеРеквизита, ЗначениеРеквизита);
		КонецЕсли;
			
		
	КонецЦикла;  
	
	Возврат ОбщиеСведения;
КонецФункции

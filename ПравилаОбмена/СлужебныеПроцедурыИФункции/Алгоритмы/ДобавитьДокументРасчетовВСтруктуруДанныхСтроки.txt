Процедура ДобавитьДокументРасчетовВСтруктуруДанныхСтроки(ДанныеXDTO, Строка, СтруктураДанныхСтроки)
	Если СтруктураДанныхСтроки.Свойство("ДокументРасчетов")
		И ЗначениеЗаполнено(СтруктураДанныхСтроки.ДокументРасчетов)
		И СтруктураДанныхСтроки.ДокументРасчетов.Свойство("ТипЗначения")
		И СтруктураДанныхСтроки.ДокументРасчетов.ТипЗначения = "ИсходныйДокумент" Тогда
		ПервичныйДокумент = СтруктураДанныхСтроки.ДокументРасчетов;
		ИмяПКОПервичныйДокумент = "Документ_ПервичныйДокумент_Получение";
	Иначе
		ПервичныйДокумент = ПодготовитьСтруктуруДанныхПервичногоДокумента(ДанныеXDTO, Строка);	
		ИмяПКОПервичныйДокумент = "Документ_ПервичныйДокументИзСтруктуры_Получение";
	КонецЕсли;
	
	ИнструкцияПервичныйДокумент = Новый Структура("Значение, ИмяПКО",
		ПервичныйДокумент, ИмяПКОПервичныйДокумент);
	СтруктураДанныхСтроки.Вставить("ДокументРасчетов", ИнструкцияПервичныйДокумент);
КонецПроцедуры

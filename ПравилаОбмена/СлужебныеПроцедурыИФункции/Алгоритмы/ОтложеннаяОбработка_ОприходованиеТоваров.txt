Процедура ОтложеннаяОбработка_ОприходованиеТоваров(Объект, КомпонентыОбмена)
	ЗаполнитьСтатусыУказанияСерий(Объект, КомпонентыОбмена);
	Если НЕ ЗначениеЗаполнено(Объект.Валюта) Тогда
		Если ЗначениеЗаполнено(Объект.ВидЦены) И ОбщегоНазначения.СсылкаСуществует(Объект.ВидЦены) Тогда
			ВалютаЦены = ОбщегоНазначения.ЗначениеРеквизитаОбъекта(Объект.ВидЦены, "ВалютаЦены");
			Объект.Валюта = ВалютаЦены;
		КонецЕсли;
		Если НЕ ЗначениеЗаполнено(Объект.Валюта) Тогда
			Объект.Валюта = ВалютаРегламентированногоУчета(КомпонентыОбмена);
		КонецЕсли;
	КонецЕсли;             
	 
	Если НЕ ЗначениеЗаполнено(Объект.СтатьяДоходов) Тогда          
		СтатьяДоходов = ПланыВидовХарактеристик.СтатьиДоходов.НайтиПоНаименованию("Статья приходов по умолчанию");
		Объект.СтатьяДоходов = СтатьяДоходов;
	КонецЕсли;
КонецПроцедуры

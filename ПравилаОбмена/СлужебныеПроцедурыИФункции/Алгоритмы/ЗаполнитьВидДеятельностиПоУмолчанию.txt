Процедура ЗаполнитьВидДеятельностиПоУмолчанию(Объект, ИмяРеквизитаДляЗаполнения)
	Если Не ЗначениеЗаполнено(Объект[ИмяРеквизитаДляЗаполнения]) Тогда
		//И ОбщегоНазначения.СсылкаСуществует(Объект[ИмяРеквизитаДляЗаполнения]) 
		МенеджерДокумента = ОбщегоНазначения.МенеджерОбъектаПоПолномуИмени(Объект.Метаданные().ПолноеИмя());
		
		Если ТипЗнч(Объект) = Тип("ДокументОбъект.ВводОстатков") Тогда
			ПараметрыЗаполнения = МенеджерДокумента.ИнициализироватьПараметрыВидовДеятельностиНДС(Объект);
		ИначеЕсли ТипЗнч(Объект) = Тип("ДокументОбъект.ПеремещениеТоваров")
			ИЛИ ТипЗнч(Объект) = Тип("ДокументОбъект.ЗаказНаПеремещение") Тогда
			ПараметрыЗаполнения = УчетНДСУПКлиентСервер.ПараметрыЗаполненияВидаДеятельностиНДС();
			ПараметрыЗаполнения.Организация              = Объект.Организация;
			ПараметрыЗаполнения.Дата                     = Объект.Дата;
			ПараметрыЗаполнения.Склад                    = Объект.СкладПолучатель;
			ПараметрыЗаполнения.ДвижениеТоваровНаСкладах = Истина;
		Иначе	
			ПараметрыЗаполнения = МенеджерДокумента.ПараметрыЗаполненияВидаДеятельностиНДС(Объект);
		КонецЕсли;
		
		УчетНДСУП.ЗаполнитьВидДеятельностиНДС(Объект[ИмяРеквизитаДляЗаполнения], ПараметрыЗаполнения);
	КонецЕсли;
КонецПроцедуры

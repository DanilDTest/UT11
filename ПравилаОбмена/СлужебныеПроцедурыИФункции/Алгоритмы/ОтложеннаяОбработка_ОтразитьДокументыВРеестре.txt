Процедура ОтложеннаяОбработка_ОтразитьДокументыВРеестре(КомпонентыОбмена)
	Возврат;
	
	// Необходимо отразить в РеестреДокументов непроведенные документы.
	
	// Сформируем ключи реестра документов.
	Справочники.КлючиРеестраДокументов.СоздатьОбновитьКлючиРеестра();
	
	ДокументыДляОтложенногоПроведения = КомпонентыОбмена.ДокументыДляОтложенногоПроведения.Скопировать();
	ДокументыДляОтложенногоПроведения.Свернуть("ДокументСсылка");
	
	Если ДокументыДляОтложенногоПроведения.Количество() = 0 Тогда
		Возврат;
	КонецЕсли;
	
	
	// Получим типы документов для отложенного проведения, содержащихся в РеестреДокументов.
	ДокументыДляОтраженияВРеестре = Новый Соответствие;
	Для Каждого СтрокаТаблицы Из ДокументыДляОтложенногоПроведения Цикл
	
		Если ЗначениеЗаполнено(СтрокаТаблицы.ДокументСсылка)
			И ОбщегоНазначения.СсылкаСуществует(СтрокаТаблицы.ДокументСсылка) 
			И РегистрыСведений.РеестрДокументов.ОбъектВключенВСоставДанныхРеестра(СтрокаТаблицы.ДокументСсылка) Тогда
		
			ДокументыДляОтраженияВРеестре.Вставить(СтрокаТаблицы.ДокументСсылка.Метаданные());
		КонецЕсли;
		
	КонецЦикла;
	
	Если ДокументыДляОтраженияВРеестре.Количество() > 0 Тогда
		РегистрыСведений.РеестрДокументов.ОтразитьДанныеДокументовВРеестре(ДокументыДляОтраженияВРеестре);
	КонецЕсли;
КонецПроцедуры

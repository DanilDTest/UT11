	Если ПараметрыУказанияСерий.ТоварВШапке Тогда
		Если ЗначениеЗаполнено(Объект.Серия)
			И НоменклатураКлиентСервер.ВЭтомСтатусеСерииУказываютсяВТЧСерии(Объект.СтатусУказанияСерий, ПараметрыУказанияСерий) Тогда
			
			СтрокаСерии = Объект[ПараметрыУказанияСерий.ИмяТЧСерии].Добавить();
			ЗаполнитьЗначенияСвойств(СтрокаСерии, Объект);
		КонецЕсли;
	Иначе
		Для Каждого СтрокаТЧ Из Объект[ПараметрыУказанияСерий.ИмяТЧТовары] Цикл
			Если ЗначениеЗаполнено(СтрокаТЧ.Серия)
				И НоменклатураКлиентСервер.ВЭтомСтатусеСерииУказываютсяВТЧСерии(СтрокаТЧ.СтатусУказанияСерий, ПараметрыУказанияСерий) Тогда
				
				СтрокаСерии = Объект[ПараметрыУказанияСерий.ИмяТЧСерии].Добавить();
				ЗаполнитьЗначенияСвойств(СтрокаСерии, СтрокаТЧ);
			КонецЕсли;
		КонецЦикла;
	КонецЕсли;

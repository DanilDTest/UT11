Процедура ОтложеннаяОбработка_ОтчетОРозничныхПродажах(Объект, КомпонентыОбмена)
	Если НЕ ЗначениеЗаполнено(Объект.Валюта) Тогда
		Объект.Валюта = КомпонентыОбмена.ПараметрыКонвертации.ВалютаРегламентированногоУчета;    
	КонецЕсли;               
	
	Если RM_ОбщийМодуль.ЭтоСкладКазахстана(Объект.Склад) 
		И НЕ (ЗначениеЗаполнено(Объект.Подразделение) ИЛИ ЗначениеЗаполнено(Объект.RM_Подразделение)) Тогда
	   
	    ПодразделениеСклада = ПодразделениеСклада(Объект.Склад);
		Объект.Подразделение = ПодразделениеСклада; 
		Объект.RM_Подразделение = ПодразделениеСклада;
	
	
	КонецЕсли;
	
	Объект.ВидЦены = Объект.Склад.РозничныйВидЦены;
	
	Если Не ЗначениеЗаполнено(Объект.КассоваяСмена) Тогда
		НоваяСтрока = КомпонентыОбмена.ПараметрыКонвертации.ДокументыСПодарочнымиСертификатами.Добавить();
		НоваяСтрока.ДокументСсылка   = Объект.Ссылка;
		НоваяСтрока.ДатаДокумента    = Объект.Дата;
		НоваяСтрока.ОчередьДокумента = 1;
		
		ЗаполнитьЗначенияСвойств(НоваяСтрока, Объект, "КассаККМ, Организация, Валюта");
	КонецЕсли;
	
	ЗаполнитьСтатусыУказанияСерий(Объект, КомпонентыОбмена);
	ЗаполнитьПартнерРПВТЧ(Объект);
КонецПроцедуры

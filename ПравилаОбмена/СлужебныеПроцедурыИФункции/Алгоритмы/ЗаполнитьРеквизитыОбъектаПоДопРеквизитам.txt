Процедура ЗаполнитьРеквизитыОбъектаПоДопРеквизитам(ДанныеXDTO, Объект)
	ДополнительныеРеквизиты = Неопределено;
	ДанныеXDTO.Свойство("ДополнительныеРеквизиты", ДополнительныеРеквизиты);
	  Если ДополнительныеРеквизиты = Неопределено Тогда
		Возврат; 
	КонецЕсли;
	
	МассивСтрокДляУдаления = Новый Массив;
	СтруктураДополнительныхРеквизитов = Новый Структура;   
	
	Для каждого СтрокаДопРеквизита из ДополнительныеРеквизиты Цикл 
		  
		НаименованиеСвойства = СтрокаДопРеквизита.Свойство.Наименование;	  
		Если НЕ СтрНайти(НаименованиеСвойства, "Реквизит") = 0 Тогда 
			МассивСтрокДляУдаления.Добавить(СтрокаДопРеквизита);
			
			НаименованиеРеквизита = СтрЗаменить(НаименованиеСвойства, "Реквизит", "");
			ЗначениеРеквизита = СтрокаДопРеквизита.ЗначениеСвойства;
			
			СтруктураДополнительныхРеквизитов.Вставить(НаименованиеРеквизита, ЗначениеРеквизита);
		КонецЕсли;
			
		
	КонецЦикла;  
	
	Для каждого СтрокаДляУдаления из МассивСтрокДляУдаления Цикл 
		ДополнительныеРеквизиты.Удалить(СтрокаДляУдаления);
	КонецЦикла;
	
	ЗаполнитьЗначенияСвойств(Объект, СтруктураДополнительныхРеквизитов);
КонецПроцедуры

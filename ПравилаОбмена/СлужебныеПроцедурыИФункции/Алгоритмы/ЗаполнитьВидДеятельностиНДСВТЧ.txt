Процедура ЗаполнитьВидДеятельностиНДСВТЧ(КомпонентыОбмена, ДанныеИБ, Таблица, ИмяКолонки, ИмяРеквизита = "", ПриобретениеТоваров = Истина)
	Если ВерсияФорматаЧислом(КомпонентыОбмена.ВерсияФорматаОбмена) >= ВерсияФорматаЧислом("1.6") Тогда
		
		Если ИмяРеквизита = "" Тогда
			// Получим вид деятельности НДС по умолчанию.
			СтруктураДанныеИБ = Новый Структура("Организация, Дата, Склад");
			ЗаполнитьЗначенияСвойств(СтруктураДанныеИБ, ДанныеИБ);
			
			ПараметрыУчетаПоОрганизации = УчетНДСУП.ПараметрыУчетаПоОрганизации(
					СтруктураДанныеИБ.Организация,
					СтруктураДанныеИБ.Дата,
					СтруктураДанныеИБ.Склад);
			
			Если ПриобретениеТоваров Тогда
				ВидДеятельностиНДС = ПараметрыУчетаПоОрганизации.ОсновнойВидДеятельностиНДСЗакупки;		
			Иначе
				ВидДеятельностиНДС = ПараметрыУчетаПоОрганизации.ОсновноеНалогообложениеНДСПродажи;
			КонецЕсли;
			 
		Иначе
			ВидДеятельностиНДС = ДанныеИБ[ИмяРеквизита];
		КонецЕсли;
		
		Таблица.Колонки.Добавить(ИмяКолонки);
		Таблица.ЗаполнитьЗначения(ВидДеятельностиНДС, ИмяКолонки);	
	КонецЕсли;
КонецПроцедуры

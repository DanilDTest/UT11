Процедура ЗагрузитьБанкВБанковскийСчет(ДанныеXDTO, ПолученныеДанные, КомпонентыОбмена)
	Если НЕ ЗначениеЗаполнено(ПолученныеДанные.Банк)
		И ДанныеXDTO.КлючевыеСвойства.Свойство("Банк") Тогда
		
		Если ДанныеXDTO.КлючевыеСвойства.Банк.Свойство("БИК") Тогда
			ПолученныеДанные.БИКБанка = ДанныеXDTO.КлючевыеСвойства.Банк.БИК;
		КонецЕсли;
		
		Если ДанныеXDTO.КлючевыеСвойства.Банк.Свойство("СВИФТБИК") Тогда
			ПолученныеДанные.СВИФТБанка = ДанныеXDTO.КлючевыеСвойства.Банк.СВИФТБИК;
		КонецЕсли;
	
		Если ДанныеXDTO.КлючевыеСвойства.Банк.Свойство("Наименование") Тогда
			ПолученныеДанные.НаименованиеБанка = ДанныеXDTO.КлючевыеСвойства.Банк.Наименование;
		КонецЕсли;
		
		Если ДанныеXDTO.КлючевыеСвойства.Банк.Свойство("КоррСчет") Тогда
			ПолученныеДанные.КоррСчетБанка = ДанныеXDTO.КлючевыеСвойства.Банк.КоррСчет;
		КонецЕсли;
		
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(ПолученныеДанные.БанкДляРасчетов)
		И ДанныеXDTO.КлючевыеСвойства.Свойство("БанкДляРасчетов") Тогда
		
		Если ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.Свойство("БИК") Тогда
			ПолученныеДанные.БИКБанкаДляРасчетов = ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.БИК;
		КонецЕсли;
		
		Если ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.Свойство("СВИФТБИК") Тогда
			ПолученныеДанные.СВИФТБанкаДляРасчетов = ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.СВИФТБИК;
		КонецЕсли;
	
		Если ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.Свойство("Наименование") Тогда
			ПолученныеДанные.НаименованиеБанкаДляРасчетов = ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.Наименование;
		КонецЕсли;
		
		Если ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.Свойство("КоррСчет") Тогда
			ПолученныеДанные.КоррСчетБанкаДляРасчетов = ДанныеXDTO.КлючевыеСвойства.БанкДляРасчетов.КоррСчет;
		КонецЕсли;
					
	КонецЕсли;
КонецПроцедуры

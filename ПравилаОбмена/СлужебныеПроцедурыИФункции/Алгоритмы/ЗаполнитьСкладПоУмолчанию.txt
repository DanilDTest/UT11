Процедура ЗаполнитьСкладПоУмолчанию(ПолученныеДанные, ДанныеИБ, КомпонентыОбмена, ИмяРеквизита = "Склад")
	Если ЗначениеЗаполнено(ПолученныеДанные[ИмяРеквизита]) Тогда
		Возврат;
	ИначеЕсли ДанныеИБ <> Неопределено И ЗначениеЗаполнено(ДанныеИБ[ИмяРеквизита]) Тогда
		// Склад был заполнен в данных ИБ - поместим его в ПолученныеДанные чтобы не затерлось.
		ПолученныеДанные[ИмяРеквизита] = ДанныеИБ[ИмяРеквизита];
		Возврат;
	КонецЕсли;
	
	ПолученныеДанные[ИмяРеквизита] = СкладДляЗагрузки(ПолученныеДанные[ИмяРеквизита], КомпонентыОбмена.ПараметрыКонвертации);

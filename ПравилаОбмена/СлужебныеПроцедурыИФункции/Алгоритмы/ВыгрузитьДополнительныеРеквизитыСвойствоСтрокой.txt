Процедура ВыгрузитьДополнительныеРеквизитыСвойствоСтрокой(СтруктураДополнительныхРеквизитов, ДанныеXDTO)
	ДополнительныеРеквизиты = Неопределено;
	ДанныеXDTO.Свойство("ДополнительныеРеквизиты", ДополнительныеРеквизиты);
	
	Если ДополнительныеРеквизиты = Неопределено Тогда
		ДополнительныеРеквизиты = Новый ТаблицаЗначений;
		ДополнительныеРеквизиты.Колонки.Добавить("Свойство");
		ДополнительныеРеквизиты.Колонки.Добавить("ЗначениеСвойства");   
	КонецЕсли;
	 
	 
	Для каждого ДопРеквизит из СтруктураДополнительныхРеквизитов Цикл
	 	НаименованиеРеквизита = СтрШаблон("Реквизит%1", ДопРеквизит.Ключ);
		Значение = ДопРеквизит.Значение;
			
		СтруктураСвойства = Новый Структура;
		СтруктураСвойства.Вставить("Наименование", НаименованиеРеквизита);    
		
		ТипЗначенияСвойства = ТипЗнч(Значение);
		ИмяПКО = "";
		ИмяПКОГруппа = "";
		Если ТипЗначенияСвойства = Тип("Число") 
			ИЛИ ТипЗначенияСвойства = Тип("Строка")
			ИЛИ ТипЗначенияСвойства = Тип("Булево") 
			ИЛИ ТипЗначенияСвойства = Тип("Структура")
			ИЛИ ТипЗначенияСвойства = Тип("Дата") Тогда    
			
				НовСтрока = ДополнительныеРеквизиты.Добавить();
				НовСтрока.Свойство = СтруктураСвойства;
				НовСтрока.ЗначениеСвойства = Значение;
				Продолжить;      
		КонецЕсли;
			
		ВыгрузкаДопРеквизитов_ОпределитьИмяПКО(Значение, ИмяПКО, ИмяПКОГруппа); 
		
		Если ИмяПКО = "" Тогда
			Продолжить;
		КонецЕсли; 	      
		
		Если ИмяПКОГруппа <> "" И Значение.ЭтоГруппа Тогда
			ТекИмяПКО = ИмяПКОГруппа;
		Иначе
			ТекИмяПКО = ИмяПКО;
		КонецЕсли;           
		
		
	 	НовСтрока = ДополнительныеРеквизиты.Добавить();
		НовСтрока.Свойство = СтруктураСвойства;
		НовСтрока.ЗначениеСвойства = Новый Структура("Значение, ИмяПКО", Значение, ТекИмяПКО);
	
	КонецЦикла; 
	                    
	ДанныеXDTO.Вставить("ДополнительныеРеквизиты",  ДополнительныеРеквизиты);
КонецПроцедуры

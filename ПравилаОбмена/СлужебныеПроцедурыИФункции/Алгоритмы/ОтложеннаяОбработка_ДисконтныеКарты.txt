Процедура ОтложеннаяОбработка_ДисконтныеКарты(Объект, КомпонентыОбмена, ОбъектМодифицирован)
	ЗаполнениеПартнераВШапкеДокумента(Объект);
	
	Если НЕ ЗначениеЗаполнено(Объект.Статус) Тогда
		Объект.Статус = Перечисления.СтатусыКартЛояльности.Действует;
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(Объект.Наименование) Тогда
		Объект.Наименование = Строка(Объект.Владелец)
		               + ?(ЗначениеЗаполнено(Объект.Штрихкод), " " + Строка(Объект.Штрихкод), "")
		               + ?(ЗначениеЗаполнено(Объект.МагнитныйКод), " " + Строка(Объект.МагнитныйКод), "");
	КонецЕсли;
	
	Если НЕ ЗначениеЗаполнено(Объект.Владелец) Тогда
		Объект.Владелец = ЭлементПоУмолчанию(КомпонентыОбмена, "ВидКартЛояльности");
	КонецЕсли;
КонецПроцедуры

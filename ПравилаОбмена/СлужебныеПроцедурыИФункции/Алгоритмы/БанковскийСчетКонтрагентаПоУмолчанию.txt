Функция БанковскийСчетКонтрагентаПоУмолчанию(Контрагент, Валюта)
	Запрос = Новый Запрос("ВЫБРАТЬ ПЕРВЫЕ 2 Ссылка КАК Ссылка
		|	ИЗ Справочник.БанковскиеСчетаКонтрагентов
		|	ГДЕ Владелец = &Контрагент 
		|		И ВалютаДенежныхСредств = &Валюта 
		|		И НЕ Закрыт 
		|		И НЕ ПометкаУдаления");
	
	Запрос.УстановитьПараметр("Контрагент", Контрагент);
	Запрос.УстановитьПараметр("Валюта", Валюта);
	
	БанкСчета = Запрос.Выполнить().Выгрузить();
	
	Если БанкСчета.Количество() = 1 Тогда
		Возврат БанкСчета[0].Ссылка;
	Иначе
		Возврат Справочники.БанковскиеСчетаКонтрагентов.ПустаяСсылка();
	КонецЕсли;
КонецФункции

		И ЗначениеЗаполнено(ДанныеXDTO[ИмяСвойстваXDTO]) Тогда
		
		Если ДанныеXDTO[ИмяСвойстваXDTO].Свойство("Ссылка") Тогда
			ИмяПКО = "Справочник_Валюты";
		Иначе
			ИмяПКО = "Справочник_Валюты_ИзСтруктуры";
		КонецЕсли;

	КонецЕсли;

Процедура ЗаполнитьНалогообложениеНДСПоПолученнымДанным(КомпонентыОбмена, ПолученныеДанные, ДанныеXDTO, НДСПродавца = Истина)
	// Заполняем НалогообложениеНДС в документе по полученным из формата данным.
	Если ДанныеXDTO.Свойство("Налогообложение")
		И ЗначениеЗаполнено(ДанныеXDTO.Налогообложение) Тогда
		
		Если ВерсияФорматаЧислом(КомпонентыОбмена.ВерсияФорматаОбмена) >= ВерсияФорматаЧислом("1.6") Тогда
			Если НДСПродавца Тогда
				ИмяПКО = "НалогообложениеНДСПродавца";
			Иначе
				ИмяПКО = "НалогообложениеНДС";
			КонецЕсли;
			
			ПолученныеДанные.ДополнительныеСвойства.Вставить("НалогообложениеНДС",
				Новый Структура("Значение, ИмяПКО", ДанныеXDTO.Налогообложение, ИмяПКО));
		Иначе
		
			ПолученныеДанные.ДополнительныеСвойства.Вставить("НалогообложениеНДС",
				Новый Структура("Значение, ИмяПКО", ДанныеXDTO.Налогообложение, "Перечисления_ТипыНалогообложенияНДС"));
		КонецЕсли;
	ИначеЕсли ДанныеXDTO.Свойство("НалогообложениеНДСПродавца")
		И ЗначениеЗаполнено(ДанныеXDTO.НалогообложениеНДСПродавца) Тогда
		ПолученныеДанные.ДополнительныеСвойства.Вставить("НалогообложениеНДС",
			Новый Структура("Значение, ИмяПКО", ДанныеXDTO.НалогообложениеНДСПродавца, "НалогообложениеНДСПродавца"));
	КонецЕсли;

Функция ДокументыОснованияСуществуют(СчетФактура)
	ДокументыОснованияСуществуют = Ложь;
		
	Если СчетФактура.ДокументыОснования.Количество() > 0 Тогда
		
		ДокументыОснованияСуществуют = Истина;
		
		Для Каждого СтрДокументыОснования Из СчетФактура.ДокументыОснования Цикл
			
			Если Не ОбщегоНазначения.СсылкаСуществует(СтрДокументыОснования.ДокументОснование) Тогда
				ДокументыОснованияСуществуют = Ложь;
			КонецЕсли;
			
		КонецЦикла;
		
	КонецЕсли;
	
	Возврат ДокументыОснованияСуществуют;

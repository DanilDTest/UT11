Функция ВыгрузитьПереводыРеквизитов(СсылкаНаОбъект, ДанныеXDTO)
	Запрос = Новый Запрос;
		Запрос.Текст = 
		"ВЫБРАТЬ
		|	RM_МультиязычностьПереводы.Язык КАК Язык,  
		|	RM_МультиязычностьПереводы.Язык.Код КАК ЯзыкКод,
		|	RM_МультиязычностьПереводы.Реквизит КАК Реквизит,
		|	RM_МультиязычностьПереводы.Перевод КАК Значение
		|ИЗ
		|	РегистрСведений.RM_МультиязычностьПереводы КАК RM_МультиязычностьПереводы
		|ГДЕ
		|	RM_МультиязычностьПереводы.СсылкаНаОбъект = &СсылкаНаОбъект";
		
		Запрос.УстановитьПараметр("СсылкаНаОбъект", СсылкаНаОбъект);	
		ТаблицаПереводов = Запрос.Выполнить().Выгрузить();
	                                                                
		Если ТаблицаПереводов.Количество() > 0 Тогда
			ДанныеXDTO.Вставить("Переводы", ТаблицаПереводов);
		КонецЕсли;

Процедура УстановитьПризнакЗагрузкиДанныхESB(Объект, КомпонентыОбмена, ДополнительныеСведения = Неопределено)
	ПараметрыСообщения = Новый Структура;
	ПараметрыСообщения.Вставить("RM_ЗагрузкаДанныхESB",						Истина);
	ПараметрыСообщения.Вставить("RM_ЗагрузкаДанныхESB_КодУзлаОтправителя",	КомпонентыОбмена.УзелОтправителя.Код); 
	
	ПрефиксДопСведений = "RM_ЗагрузкаДанныхESB_%1"; 
	Если Не ДополнительныеСведения = Неопределено Тогда
		Для Каждого КлючЗначение Из ДополнительныеСведения Цикл
		 	КлючДопЗначения = СтрШаблон(ПрефиксДопСведений, КлючЗначение.Ключ);
		 	ПараметрыСообщения.Вставить(КлючДопЗначения,	КлючЗначение.Значение);
		КонецЦикла;
	КонецЕсли;
	
	Для Каждого КлючЗначение Из ПараметрыСообщения Цикл
		КомпонентыОбмена.ПараметрыСообщенияESB.Вставить(КлючЗначение.Ключ,	КлючЗначение.Значение);
	 	Объект.ДополнительныеСвойства.Вставить(КлючЗначение.Ключ,			КлючЗначение.Значение);
	КонецЦикла;
КонецПроцедуры

Процедура ЗаполнитьСтатусыУказанияСерий(Объект, КомпонентыОбмена)
	ПолноеИмяОбъекта  = Объект.Метаданные().ПолноеИмя();
	МенеджерДокумента = ОбщегоНазначения.МенеджерОбъектаПоПолномуИмени(ПолноеИмяОбъекта);
	
	ПараметрыУказанияСерий = Новый ФиксированнаяСтруктура(НоменклатураСервер.ПараметрыУказанияСерий(Объект, МенеджерДокумента));
	
	Если ПолноеИмяОбъекта = "Документ.СборкаТоваров" Тогда
		ИменаТЧ = "Шапка,ТЧ";	
	ИначеЕсли ПолноеИмяОбъекта = "Документ.ПоступлениеОтПереработчика" Тогда
		ИменаТЧ = "Товары,ВозвратныеОтходы";
	ИначеЕсли ПолноеИмяОбъекта = "Документ.ПроизводствоБезЗаказа" Тогда
		ИменаТЧ = "ВыходныеИзделия,ПобочныеИзделия,МатериалыИРаботы";
	Иначе
		ИменаТЧ = "";	
	КонецЕсли;
	
	МассивИмен = СтрРазделить(ИменаТЧ, ",", Истина);
	
	Для Каждого ИмяТЧ Из МассивИмен Цикл
		Если ЗначениеЗаполнено(ИмяТЧ) Тогда
			ВремПараметры = ПараметрыУказанияСерий[ИмяТЧ];
		Иначе
			ВремПараметры = ПараметрыУказанияСерий;
		КонецЕсли;
	
		ЗаполнитьВидНоменклатурыУСерийВТЧ(Объект, ВремПараметры, КомпонентыОбмена, ПолноеИмяОбъекта);
		НоменклатураСервер.ЗаполнитьСтатусыУказанияСерий(Объект, ВремПараметры);
		ПеренестиСерииВТЧСерии(Объект, ВремПараметры, КомпонентыОбмена);
		НоменклатураСервер.ОчиститьНеиспользуемыеСерии(Объект, ВремПараметры);
	КонецЦикла;
КонецПроцедуры
